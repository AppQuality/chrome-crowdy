[
	// In the following scheme, /type\ represents the type. 

	// This stores a click event
	{
		"type": "click",
		"time": /string\,
		"data": {
			"url": /string\,
			"id_obj": /string\,
			"class_obj": /string\,
			"position_x": /string\,
			"position_y": /string\
		}
	}

	// This stores a console log
	{
		"type": "console",
		"time": /string\,
		"data": {
			"type": "c_log" OR "c_warn" OR "c_error" OR "c_debug",
			"msg":/string\
		}
	}

	// This stores a generical error (SyntaxError, UnknownType...)
	{
		"type": "error",
		"time": /string\,
		"data": {
			"message": /string\,
			"filename": /string\,
			"lineno": /number\,
			"colno": /number\,
			"error": /object\
		}
	}

	// This stores a GET error
	{
		"type": "errorget",
		"time": /string\,

			"initiator": /string\,

			"parentFrameId": /number\,
			"requestId": /string\,
			"tabId": /number\,
			"timeStamp": /number\,
			"type": /string\,

		}
